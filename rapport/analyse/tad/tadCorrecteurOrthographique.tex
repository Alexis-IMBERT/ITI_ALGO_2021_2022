\begin{tad}
	\tadNom{CorrecteurOrthographique}
	%\tadParametres{}
	\tadDependances{Mot, Dictionnaire, \booleen, \naturel, TableauDeMots, TableauBooleens, TableauDEntiers, \entier, MotsDansPhrase, TableauPositions}
	\begin{tadOperations}{CorrecteurOrthographique}
		\tadOperation{tableauDEntiersVide}%
			{\tadUnParam{            }}%
			{\tadUnParam{TableauDEntiers}}
		\tadOperation{obtenirLongueurTabEntiers}%
			{\tadUnParam{TableauDEntiers}}%
			{\tadUnParam{Naturel}}
		\tadOperation{fixerLongueurTabEntiers}%
			{\tadDeuxParams{TableauDEntiers}{\naturel}}%
			{\tadUnParam{TableauDEntiers}}
		\tadOperationAvecPreconditions{obtenirIemeEntier}%
			{\tadDeuxParams{TableauDEntiers}{\naturel}}%
			{\tadUnParam{\entier}}
		\tadOperation{obtenirLesEntiers}%
			{\tadUnParam{TableauDEntiers}}%
			{\tadUnParam{\tableauUneDimension{Taille\_Max}{d'}{\entier}}}
		\tadOperation{ajouterEntier}%
			{\tadDeuxParams{TableauDEntiers}{\entier}}%
			{\tadUnParam{TableauDEntiers}}
		\tadOperation{supprimerTableauEntiers}%
			{\tadUnParam{TableauDEntiers}}%
			{\tadUnParam{TableauDEntiers}}
		\tadOperation{motsEtPositionsVide}%
			{\tadUnParam{           }}%
			{\tadUnParam{MotsDansPhrase}}
		\tadOperation{obtenirTabMots}%
			{\tadUnParam{MotsDansPhrase}}%
			{\tadUnParam{TableauDeMots}}
		\tadOperation{obtenirTabPositions}%
			{\tadUnParam{MotsDansPhrase}}%
			{\tadUnParam{TableauPositions}}
		\tadOperation{sontPresents}%
			{\tadDeuxParams{TableauDeMots}{Dictionnaire}}%
			{\tadUnParam{TableauBooleens}}
		\tadOperation{proposerMots}%
			{\tadDeuxParams{Mot}{Dictionnaire}}%
			{\tadUnParam{TableauDeMots}}
		\tadOperation{phraseEnMots}%
			{\tadUnParam{\chaine}}%
			{\tadUnParam{MotsDansPhrase}}
	\end{tadOperations}
		
	\begin{tadSemantiques}{CorrecteurOrthographique}
		\tadSemantique{sontPresents}%
			{renvoie un tableau de Booléens indiquant la présence ou non des mots de la liste.}
		\tadSemantique{proposerMots}%
			{renvoie un tableau de mots correspondant aux corrections possibles d’un mot mal orthographi\'e.}
		\tadSemantique{phraseEnMots}%
			{renvoie un tableau contenant les mots de la phrase ainsi que leur position dans la phrase.}
	\end{tadSemantiques}
		
	\begin{tadPreconditions}{obtenirIemeEntier}
		\tadPrecondition{obtenirIemeEntier(tableauEntiers, position)}%
			{0 $\leq$ position ET position < obtenirLongueurTabEntiers(tableauEntiers)}
	\end{tadPreconditions}
		
\end{tad}
