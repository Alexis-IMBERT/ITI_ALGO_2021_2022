\begin{algorithme}
\fonction{CO\_sontPresents}%
	{mots : MOT\_TableauDeMots, d : Dictionnaire}%
	{\\CO\_TableauBooleens}%
	{i, longueur : \naturel \\ tabBool : CO\_TableauBooleens\\ mot : MOT\_mot}%
	{%
	\affecter{longueur}{MOT\_obtenirLongueurTabMots(mots)}
	\pour{i}{0}{longueur}{}%
	{	
		\affecter{mot}{MOT\_obtenirIemeMot(mots,i)}
		\sialorssinon{DICTIONNAIRE\_estPresent(dictionnaire, mot)}%
		{
			\affecter{tabBool[i]}{tVRAI)}
		}%
		{
			\affecter{tabBool[i]}{FAUX)}%
			}
	}%
	\retourner{tabBool}
	}\\
\end{algorithme}

\begin{algorithme}
\fonction{CO\_proposerMots}%
	{m : MOT\_Mot, dictionnaire : DICTIONNAIRE\_Dictionnaire}%
	{\\MOT\_TableauDeMots}%
	{resultatMots : MOT\_TableauDeMots\\ motCorrige : MOT\_Mot\\ motCorriges : MOT\_DeuxMot\\ lettre, tiret : \caractere \\ i : \naturel}%
	{%
	\pour{lettre}{'a'}{'z'}{}%
		{\pour{i}{0}{MOT\_longueurMot(m)}{}%
			{
			\affecter{motCorrige}{MOT\_remplacerLettre(m, i, lettre)}
			\sialors{DICTIONNAIRE\_estPresent(dictionnaire, motCorrige)}%
				{\instruction{MOT\_ajouterMot(resultatMots, motCorrige)}}
			}
		\pour{i}{0}{MOT\_longueurMot(m) + 1}{}%
			{\affecter{motCorrige}{MOT\_insererLettre(m, i, lettre)}
			\sialors{DICTIONNAIRE\_estPresent(dictionnaire, motCorrige)}%
				{\instruction{MOT\_ajouter(resultatMots, motCorrige)}}
			}
		}
		
	\affecter{tiret}{"-"}
	{\pour{i}{0}{MOT\_longueurMot(m)}{}%
		{
		\affecter{motCorrige}{MOT\_remplacerLettre(m, i, tiret)}
		\sialors{DICTIONNAIRE\_estPresent(dictionnaire, motCorrige)}%
			{\instruction{MOT\_ajouterMot(resultatMots, motCorrige)}}
		}
	\pour{i}{0}{MOT\_longueurMot(m) + 1}{}%
		{
		\affecter{motCorrige}{MOT\_insererLettre(m, i, tiret)}
		\sialors{DICTIONNAIRE\_estPresent(dictionnaire, motCorrige)}%
			{\instruction{MOT\_ajouter(resultatMots, motCorrige)}}
		}
		}\\
	
	\pour{i}{0}{MOT\_longueurMot(m)}{}%
		{\affecter{motCorrige}{MOT\_supprimerLettre(m, i)}
		\sialors{DICTIONNAIRE\_estPresent(motCorrige, d)}%
			{\instruction{MOT\_ajouter(resultatMots, motCorrige)}}
		}
		
	\pour{i}{1}{MOT\_longueurMot(m) - 1}{}%
		{\affecter{motCorriges}{MOT\_decomposerMot(m, i)}
		\sialors{DICTIONNAIRE\_estPresent(dictionnaire, motCorriges.mot1)}%
			{\instruction{MOT\_ajouter(resultatMots, motCorriges.mot1)}}
		\sialors{DICTIONNAIRE\_estPresent(dictionnaire, motsCorriges.mot2)}%
			{\instruction{MOT\_ajouter(resultatMots, motCorriges.mot2)}}
		}
	
	\retourner{resultatMots}
	}\\
\end{algorithme}

\begin{algorithme}
\fonction{CO\_phraseEnMots}%
	{phrase : \caractere}%
	{CO\_MotsDansPhrase}%
	{i, pos, longueurPhrase : \naturel \\
    	temp : \chaine \\
    	mot : MOT\_Mot\\
   	motsPhrase : CO\_MotsDansPhrase\\
	dansUnMot : \booleen \\
	tabMots : MOT\_TableauDeMots\\
	tabPos : CO\_TableauPositions}%
	{
	\affecter{pos}{0}
	\affecter{longueurPhrase}{longueur(phrase)}
	\affecter{motsPhrase}{CO\_motsEtPositionsVide()}
	\affecter{tabMots}{CO\_obtenirTabMots(motsPhrase)}
	\affecter{tabPos}{CO\_obtenirTabPositions(motsPhrase)}
	\affecter{dansUnMot}{FAUX}
	
	\pour{i}{0}{MOT\_longueurMot(m)}{}%
		{
		\sialorssinon{DICTIONNAIRE\_estPresent(motCorrige, d)}%
			{
			\sialors{NON(dansUnMot)}%
			{
				\affecter{pos}{i}
				\affecter{dansUnMot}{VRAI}
			}
			
			\affecter{temp[i - pos]}{phrase[i]}
			}%
			{
			\sialors{dansUnMot}%
			{
				\affecter{dansUnMot}{FAUX}
				\affecter{mot}{MOT\_creerMot(temp)}
				\instruction{MOT\_ajouterMot(tabMots, mot)}
				\instruction{CO\_ajouterEntier(tabPos, pos)}
			}
			
			
			
			}
		}





	\retourner{motsPhrase}
	}
\end{algorithme}
